    def update(self, ingreso, num_hc, apellido):

                archivo.write(
                    f"[{datetime.now().strftime('%Y-%m-%d %H:%M:%S')}] "
                    f"Alta registrada en fecha {ingreso}. Paciente H.C. {num_hc}, apellido: {apellido}\n")
