Código Auxiliar: Conexión al Servidor

Este es el código del servidor que recibe información mediante el decorador indicador_baja utilizando sockets.

# servidor.py
import socket

'''
Estando encendido en otra máquina, cuando se borra un registro de la base de datos 
recibe esa información y la registra en el archivo backup_datos_borrados.txt 
ubicado en la máquina donde está el servidor. 
'''

serversocket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
serversocket.bind(("", 456))
serversocket.listen(5)
print("Servidor activo, esperando...")

while True:
    connection, address = serversocket.accept()
    data = connection.recv(1024).decode("utf-8")
    if data:
        with open("backup_datos_borrados.txt", "a", encoding="utf-8") as archivo:
            archivo.write(data + "\n")
        print(f"Log recibido de {address}: {data}")
        
    connection.close()
